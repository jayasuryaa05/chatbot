  <input type="text" id="userInput" placeholder="Type a message..." onkeydown="if(event.key==='Enter') sendMsg()" />
  <button onclick="sendMsg()">Send</button>

  <script>
    function sendMsg() {
      let userText = document.getElementById("userInput").value;
      let chatbox = document.getElementById("chatbox");

      chatbox.innerHTML += "<div class='msg user'>" + userText + "</div>";
      document.getElementById("userInput").value = "";

      fetch("/get?msg=" + userText)
        .then(response => response.text())
        .then(botReply => {
          chatbox.innerHTML += "<div class='msg bot'>" + botReply + "</div>";
        });
    }
  </script>
